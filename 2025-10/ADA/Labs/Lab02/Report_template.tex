\section{Algorithm Solution}
Here, you should include the pseudo-code that solves the problem. For instance:

\begin{algorithm}[H]
    \begin{algorithmic}[1]
        \Require{ $n\in\mathbb{N}$, the number of disks to move.}
        \Require{$t \equiv \left(s, e, o \right)$, a tuple with the three robs (\textbf{\emph{\underline{s}}}tart, \textbf{\emph{\underline{e}}}nd and \textbf{\emph{\underline{o}}}ther)}

        \Procedure{SolveTowersOfHanoi}{$n, t \equiv \left(s, e, o \right)$}
            \If{$n = 1$}
                \State \Return $\left\langle t \right\rangle$
            \Else
                \State $M \leftarrow \emptyset$
                \State $M \leftarrow M \bigcup \text{\Call{SolveTowersOfHanoi}{\ensuremath{n - 1, \left(s, e, o\right)}}}$
                \State $M \leftarrow M \bigcup \text{\Call{SolveTowersOfHanoi}{\ensuremath{1, \left(s, e, o\right)}}}$
                \State $M \leftarrow M \bigcup \text{\Call{SolveTowersOfHanoi}{\ensuremath{n - 1, \left(s, e, o\right)}}}$
                \State \Return $M$
            \EndIf
            \EndProcedure
    \end{algorithmic}
\end{algorithm}
